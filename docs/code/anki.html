import genanki
import os
# 
Define an Anki flashcard model with fields for questions and answers
Suitable for use in the Anki flashcard application

def get_model():
# 
    my_model = genanki.Model(
        1607392319,
        'Anki Model',
        fields=[
            {'name': 'Question'},
            {'name': 'Answer'},
        ],
        templates=[
            {
                'name': 'Card',
                'qfmt': '{{Question}}',
                'afmt': '{{FrontSide}}<hr id="answer">{{Answer}}',
            },
        ])
    return my_model
# 
Define and initialize an anki deck, where we can add cards.

Returns
------
anki deck

def get_deck(deck_name):
# 
    my_deck = genanki.Deck(
        2059400110,
        deck_name)
    return my_deck
# 
Create a card for a question, answer pair.

Returns
------
anki card

def add_question(question, answer, curr_model):
# 
    my_note = genanki.Note(
        model=curr_model,
        fields=[question, answer])
    return my_note
# 
Create a package for a deck

Returns
------
None

def add_package(deck, output_fname):
# 
    app_cwd = os.path.dirname(os.path.realpath(__file__))
    anki_path = os.path.join(app_cwd, "anki_decks")
    if not os.path.exists(anki_path):
        os.makedirs(anki_path)
    genanki.Package(deck).write_to_file(f'{anki_path}/{output_fname}.apkg')
